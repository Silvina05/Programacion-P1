#include <stdio.h>

void ordenarBurbuja(int notas[], int n);

int main() {
    int notas[5];
    int n=5;
    int acum=0;

    // Cargar
    for(int i = 0; i < 5; i++) {
        do {
        printf("Ingrese la nota del alumno (del 0-10) : ");
        scanf("%d", &notas[i]);
        }while (notas[i]<0 || notas[i]>10);
        acum=acum+notas[i];
    }

    // Mostrar
    printf("las notas son: ");
    for(int i = 0; i < 5; i++) {
        printf("%d \n", notas[i]);
    }


    ordenarBurbuja( notas,  n);

    printf ("la nota mas baja es: %d\n", notas[0]);
    printf ("la nota mas alta es: %d\n", notas[4]);
    printf ("la nota media es: %d\n", acum/n);


    return 0;
}
void ordenarBurbuja(int notas[], int n) {
    int i, j, aux;
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (notas[j] > notas[j + 1]) {
                // Intercambio
                aux = notas[j];
                notas[j] = notas[j + 1];
                notas[j + 1] = aux;
            }
        }
    }
}
