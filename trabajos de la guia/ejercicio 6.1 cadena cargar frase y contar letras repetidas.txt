#include <stdio.h>

int main() {
    char frase[100];
    char letra;
    int contador = 0;

    printf("Ingrese una frase: ");
    gets(frase);

    printf("Ingrese una letra: ");
    scanf("%c", &letra);

    for (int i = 0; frase[i] != '\0'; i++) {
        if (frase[i] == letra) {
            contador++;
        }
    }

    printf("La letra '%c' aparece %d veces.\n", letra, contador);

    return 0;
}
