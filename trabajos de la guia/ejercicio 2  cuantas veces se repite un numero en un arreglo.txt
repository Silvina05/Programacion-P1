// ordenar arreglo 1 
void ordenar1(int A[], int n) {
    int i, j, aux;
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (A[j] < A[j + 1]) {
                // Intercambio
                aux = A[j];
                A[j] = A[j + 1];
                A[j + 1] = aux;
            }
        }
    }
}


void  ordenarBurbujaDescendente(int D[], int sizeD ){
    int i, j, aux;
    for (i = 0; i < sizeD  - 1; i++) {
        for (j = 0; j < sizeD  - 1 - i; j++) {
            if (D[j] < D[j + 1]) {
                // Intercambio
                aux = D[j];
                D[j] = D[j + 1];
                D[j + 1] = aux;
            }
        }
    }
}

int busqueda (int D[], int sizeD , int buscado) {
    int acum =0;
    for (int i = 0; i < sizeD; i++) {
        if (D[i] == buscado) {
            acum++; // ¡Lo encontramos!
        }
    }
  return acum;  
}
