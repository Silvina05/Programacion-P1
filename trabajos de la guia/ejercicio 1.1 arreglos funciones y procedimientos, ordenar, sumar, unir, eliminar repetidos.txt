// ordenar arreglo 1 
void ordenar1(int A[], int n) {
    int i, j, aux;
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (A[j] < A[j + 1]) {
                // Intercambio
                aux = A[j];
                A[j] = A[j + 1];
                A[j + 1] = aux;
            }
        }
    }
}


void  ordenarBurbujaDescendente(int D[], int sizeD ){
    int i, j, aux;
    for (i = 0; i < sizeD  - 1; i++) {
        for (j = 0; j < sizeD  - 1 - i; j++) {
            if (D[j] < D[j + 1]) {
                // Intercambio
                aux = D[j];
                D[j] = D[j + 1];
                D[j + 1] = aux;
            }
        }
    }
}

int busqueda (int D[], int sizeD , int buscado) {
    int inicio = 0;
    int fin = sizeD  - 1;

    while (inicio <= fin) {
        int medio = inicio + (fin - inicio) / 2;

        if (D[medio] == buscado) {
            return 1;
        }

        if (D[medio] < buscado) {
            fin = medio - 1;// se modifica porque los numeros mayores estan a la izquierda
        } else {
            inicio = medio + 1;// intercambie con el de arriba 
        }
    }

    return 0; // No se encontró
}
