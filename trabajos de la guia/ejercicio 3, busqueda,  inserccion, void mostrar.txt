// ordenar arreglo 1 
void ordenar1(int A[], int n) {
    int i, j, aux;
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - 1 - i; j++) {
            if (A[j] > A[j + 1]) {
                // Intercambio
                aux = A[j];
                A[j] = A[j + 1];
                A[j + 1] = aux;
            }
        }
    }
}

//mostrar arreglos A y B
void mostrarArreglo(int arreglo[], int tamaño) {

        printf("%d ", arreglo[i]);
    }
    printf("\n");
}


//ordenar arreglo D con seleccion
void ordenarSeleccion(int D[],  int sizeD) {
    int i, j, min_idx, aux;
    for (i = 0; i < sizeD - 1; i++) {
        min_idx = i; // Suponemos que el actual es el mínimo
        for (j = i + 1; j < sizeD; j++) {
            if (D[j] < D[min_idx]) {
                min_idx = j; // Encontramos uno más chico
            }
        }
        // Intercambiamos el mínimo encontrado con el de la posición actual
        aux = D[min_idx];
        D[min_idx] = D[i];
        D[i] = aux;
    }


}

int inicio = 0;

        }

}
